select inventory.siteid + inventory.itemnum, inventory.minlevel, inventory.reorder, inventory.siteid from inventory
left join item on item.itemnum = inventory.itemnum where inventory.itemnum like '98%' or inventory.itemnum like '90%' or inventory.itemnum like '91%'

select pr.prnum, pr.description, pr.siteid, pr.status, prline.itemnum, prline.orderqty, pr.statusdate, prline.ponum from pr
left join prline on pr.prnum = prline.prnum and pr.siteid = prline.siteid
where pr.description like 'Generated by reorder%' 
and (prline.itemnum like '90%' or prline.itemnum like '91%' or prline.itemnum like '98%')
and prline.itemnum is not null and pr.statusdate > '9/21/2019' and len(pr.siteid) = 2
order by prline.prnum 

select po.ponum, po.description, po.siteid, po.status, po.receipts, poline.itemnum, poline.orderqty, poline.receivedqty, po.statusdate from po
left join poline on po.ponum = poline.ponum and po.siteid = poline.siteid
where (poline.itemnum like '90%' or poline.itemnum like '91%' or poline.itemnum like '98%')
and poline.itemnum is not null and len(po.siteid)=2 and po.statusdate > '9/21/2019'
order by poline.ponum 

select route, ROUTE_STOP.assetnum, ROUTE_STOP.jpnum, ROUTE_STOP.description as jp_description, ROUTE_STOP.siteid, jobtask.description as jt_description, asset.description from ROUTE_STOP 
left join jobtask on jobtask.jpnum = coalesce(route_stop.jpnum, 'BL'+route) and jobtask.siteid = 'BL'
left join asset on ROUTE_STOP.assetnum = asset.assetnum and asset.siteid = 'BL'
where ROUTE_STOP.siteid = 'BL' and not jobtask.description in ('Completion','General')
order by route 

select sparepart.assetnum, asset.description, sparepart.itemnum, item.description, quantity, sparepart.siteid from sparepart 
left join asset on asset.assetnum=sparepart.assetnum 
left join item on item.itemnum=sparepart.itemnum 
where sparepart.siteid = 'GJ' 

select invbalances.itemnum, item.description, invbalances.curbal, invcost.lastcost, issueunit, invbalances.siteid from invbalances 
left join item on invbalances.itemnum = item.itemnum 
left join invcost on invbalances.itemnum = invcost.itemnum and invbalances.location = invcost.location 
where invbalances.siteid <> 'pbm' and curbal <> 0 and invbalances.siteid <> 'ga'
