OrderVsIssueUnit
DECLARE @SITEID AS VARCHAR(5)='RAM' --quick way to change site
SELECT a.itemnum, a.vendor, a.siteid, a.issueunit AS InventoryIssueUnit, a.orderunit AS InventoryOrderUnit, b.lastdate, b.orderunit AS InvvendorOrderUnit FROM -- we only need these columns
(SELECT itemnum, vendor, siteid, issueunit, orderunit FROM inventory WHERE siteid=@SITEID) AS a -- left side of join statement, inventory
JOIN (
SELECT itemnum, vendor, lastdate, orderunit, siteid
FROM
( -- right side of join, invvender has duplicates, use row_number() to count duplicate itemnum ordered by lastdate, only return row_number = 1 entries
	SELECT invvendor.itemnum, invvendor.lastdate, invvendor.vendor, invvendor.orderunit, invvendor.siteid, 
	row_number() OVER (PARTITION BY invvendor.itemnum ORDER BY invvendor.lastdate DESC) AS rowno
	FROM invvendor 
	WHERE invvendor.siteid = @SITEID
) x
WHERE x.rowno = 1
) AS b
on a.itemnum = b.itemnum WHERE a.issueunit!=b.orderunit OR a.orderunit!=b.orderunit; -- only return entries where the units dont match
